    heap.append(item)
    i = len(heap) - 1
    if i == 0:
        return
    parent = (i - 1) // 2
    while i > 0 and heap[parent][0] > heap[i][0]:
        heap[parent], heap[i] = heap[i], heap[parent]
        i = parent
        parent = (i - 1) // 2
